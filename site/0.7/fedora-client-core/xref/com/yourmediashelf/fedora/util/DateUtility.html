41      /**
42       * <p>
43       * Regular expression string for the yearFrag:
44       * </p>
45       * <p>
46       * a numeral consisting of at least four decimal digits, optionally preceded
47       * by a minus sign; leading '0' digits are prohibited except to bring the
48       * digit count up to four.
49       * </p>
50       */
51      private final static String yearFrag = "(-?([1-9][0-9]{3,}|0[0-9]{3}))";
52  
53      /**
54       * <p>
55       * Regular expression string for the monthFrag:
56       * </p>
57       * <p>
58       * a numeral consisting of exactly two decimal digits.
59       * </p>
60       */
61      private final static String monthFrag = "(0[1-9]|1[0-2])";
62  
63      /**
64       * <p>
65       * Regular expression string for the dayFrag:
66       * </p>
67       * <p>
68       * a numeral consisting of exactly two decimal digits.
69       * </p>
70       * <p>
71       * Note that this regex does not enforce the day-of-month constraint, which
72       * states:
73       * </p>
74       * <p>
75       * The day value must be no more than 30 if month is one of 4, 6, 9, or 11;
76       * no more than 28 if month is 2 and year is not divisible 4, or is
77       * divisible by 100 but not by 400; and no more than 29 if month is 2 and
78       * year is divisible by 400, or by 4 but not by 100.
79       * </p>
80       */
81      private final static String dayFrag = "(0[1-9]|[12][0-9]|3[01])";
82  
83      /**
84       * <p>
85       * Regular expression string for the hourFrag:
86       * </p>
87       * <p>
88       * a numeral consisting of exactly two decimal digits.
89       * </p>
90       */
91      private final static String hourFrag = "([01][0-9]|2[0-3])";
92  
93      /**
94       * <p>
95       * Regular expression string for the minuteFrag:
96       * </p>
97       * <p>
98       * a numeral consisting of exactly two decimal digits.
99       * </p>
100      */
101     private final static String minuteFrag = "([0-5][0-9])";
102 
103     /**
104      * <p>
105      * Regular expression string for the secondFrag:
106      * </p>
107      * <p>
108      * a numeral consisting of exactly two decimal digits, or two decimal
109      * digits, a decimal point, and one or more trailing digits.
110      * </p>
111      */
112     private final static String secondFrag = "([0-5][0-9])(\\.([0-9]+))?";
113 
114     /**
115      * <p>
116      * Regular expression string for the endOfDayFrag:
117      * </p>
118      * <p>
119      * combines the {@link hourFrag}, {@link minuteFrag}, {@link minuteFrag},
120      * and their separators to represent midnight of the day, which is the first
121      * moment of the next day.
122      * </p>
123      */
124     private final static String endOfDayFrag = String.format(
125             "(%s:%s:%s|(24:00:00(\\.0+)?))", hourFrag, minuteFrag, secondFrag);
126 
127     /**
128      * <p>
129      * Regular expression string for the timezoneFrag:
130      * </p>
131      * <p>
132      * if present, specifies an offset between UTC and local time. Time zone
133      * offsets are a count of minutes (expressed in timezoneFrag as a count of
134      * hours and minutes) that are added or subtracted from UTC time to get the
135      * "local" time. 'Z' is an alternative representation of the time zone
136      * offset '00:00', which is, of course, zero minutes from UTC.
137      * </p>
138      */
139     private final static String timezoneFrag =
140             "(Z|(\\+|-)((0[0-9]|1[0-3]):[0-5][0-9]|14:00))?";
141 
142     /**
143      * Concatenation of the {@link yearFrag}, {@link monthFrag}, {@link dayFrag}
144      * , {@link endOfDayFrag}, {@link timezoneFrag}, and their separators.
145      */
146     private final static String combined = String.format("%s-%s-%sT%s%s",
147             yearFrag, monthFrag, dayFrag, endOfDayFrag, timezoneFrag);
148 
149     private final static Pattern XSD_DATETIME = Pattern.compile(combined);

151     private final static ConcurrentMap<String, DateTimeFormatter> formatters =

154     /**
155      * Parses lexical representations of xsd:dateTimes, e.g.
156      * "2010-01-31T14:21:03.001Z". Fractional seconds and timezone offset are
157      * optional.
158      *
159      * <p>Note: fractional seconds are only supported to three digits of
160      * precision.</p>
161      *
162      * @param input
163      *        an XML Schema 1.1 dateTime
164      * @return a DateTime representing the input
165      * @see "http://www.w3.org/TR/xmlschema11-2/#dateTime"
166      */
167     public static DateTime parseXSDDateTime(String input) {
168         Matcher m = XSD_DATETIME.matcher(input);
169         if (!m.find()) {
170             throw new IllegalArgumentException(input +
171                     " is not a valid XML Schema 1.1 dateTime.");
172         }
173 
174         int year = Integer.parseInt(m.group(1));
175         int month = Integer.parseInt(m.group(3));
176         int day = Integer.parseInt(m.group(4));
177 
178         int hour = 0, minute = 0, second = 0, millis = 0;
179         boolean hasEndOfDayFrag = m.group(11) != null;
180         if (!hasEndOfDayFrag) {
181             hour = Integer.parseInt(m.group(6));
182             minute = Integer.parseInt(m.group(7));
183             second = Integer.parseInt(m.group(8));
184             // Parse fractional seconds
185             // m.group(9), if not null/empty should be Strings such as ".5" or
186             // ".050" which convert to 500 and 50, respectively.
187             if (m.group(9) != null && !m.group(9).isEmpty()) {
188                 // parse as Double as a quick hack to drop trailing 0s.
189                 // e.g. ".0500" becomes 0.05
190                 double d = Double.parseDouble(m.group(9));
191 
192                 // Something like the following would allow for int-sized
193                 // precision, but joda-time 1.6 only supports millis (i.e. <= 999).
194                 // see: org.joda.time.field.FieldUtils.verifyValueBounds
195                 //   int digits = String.valueOf(d).length() - 2;
196                 //   fractionalSeconds = (int) (d * Math.pow(10, digits));
197                 millis = (int) (d * 1000);
198             }
199         }
200 
201         DateTimeZone zone = null;
202         if (m.group(13) != null) {
203             String tmp = m.group(13);
204             if (tmp.equals("Z")) {
205                 tmp = "+00:00";
206             }
207             zone = DateTimeZone.forID(tmp);
208         }
209         DateTime dt =
210                 new DateTime(year, month, day, hour, minute, second, millis,
211                         zone);
212 
213         if (hasEndOfDayFrag) {
214             return dt.plusDays(1);
215         }
216         return dt;
217     }

230     /**
231      * Formats a {@link DateTime} as an xsd:dateTime in canonical form.
232      *
233      * <p>Note: fractional seconds are only supported to a maximum of three
234      * digits.</p>
235      *
236      * @param dateTime
237      * @return An xsd:dateTime (in UTC) representation of date.
238      * @see "http://www.w3.org/TR/xmlschema11-2/#dateTime"

240     public static String getXSDDateTime(DateTime dateTime) {
241         return dateTime.withZone(DateTimeZone.UTC).toString(
242                 getXSDFormatter(dateTime));
243     }

245     public static DateTimeFormatter getXSDFormatter(DateTime date) {
246         int len = 0;
247         int millis = date.getMillisOfSecond();
248 
249         if (millis > 0) {
250             // 0.050 becomes .05 (up to three digits, dropping trailing 0s)
251             DecimalFormat df = new DecimalFormat(".###");
252             double d = millis / 1000.0;
253             len = String.valueOf(df.format(d)).length() - 1;
254         }
255         return getXSDFormatter(len);
256     }
257 
258     /**
259      * Returns an xsd:dateTime formatter with the specified millisecond precision.
260      *
261      * @param millisLength number of digits of millisecond precision. Currently,
262      * only 0-3 are valid arguments.
263      * @return a formatter for yyyy-MM-dd'T'HH:mm:ss[.SSS]Z
264      */
265     public static DateTimeFormatter getXSDFormatter(int millisLength) {
266         String key = String.valueOf(millisLength);
267         if (formatters.get(key) == null) {
268             DateTimeFormatterBuilder bldr =
269                     new DateTimeFormatterBuilder().appendYear(4, 9)
270                             .appendLiteral('-').appendMonthOfYear(2)
271                             .appendLiteral('-').appendDayOfMonth(2)
272                             .appendLiteral('T').appendHourOfDay(2)
273                             .appendLiteral(':').appendMinuteOfHour(2)
274                             .appendLiteral(':').appendSecondOfMinute(2);
275             if (millisLength > 0) {
276                 bldr =
277                         bldr.appendLiteral('.').appendFractionOfSecond(
278                                 millisLength, millisLength);
279             }
280             bldr = bldr.appendTimeZoneOffset("Z", true, 2, 4);
281             formatters.putIfAbsent(key, bldr.toFormatter());
282         }
283         return formatters.get(key);
284     }
