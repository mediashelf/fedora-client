20  package com.yourmediashelf.fedora.client.request;
21  
22  import java.util.regex.Pattern;
23  
24  /**
25   * Abstract base class for the *Relationship methods.
26   *
27   * @author Edwin Shin
28   */
29  public abstract class RelationshipsRequest extends
30          FedoraRequest<RelationshipsRequest> {
31  
32      protected final String pid;
33  
34      public RelationshipsRequest(String subject) {
35          pid = SubjectProcessor.getSubjectPID(subject);
36          addQueryParam("subject", SubjectProcessor.getSubjectAsUri(subject));
37      }
38  
39      /**
40       * Helper class to get pid from subject and to get URI form of subject.
41       * Subject can either be a pid or an info:fedora/ uri
42       *
43       * Adapted from {@link org.fcrepo.server.management.DefaultManagement.SubjectProcessor}
44       */
45      private static class SubjectProcessor {
46  
47          private static String FEDORA_URI_PREFIX = "info:fedora/";
48  
49          private static Pattern pidRegex =
50                  Pattern.compile("^([A-Za-z0-9]|-|\\.)+:(([A-Za-z0-9])|-|\\.|~|_|(%[0-9A-F]{2}))+$");
51  
52          static String getSubjectAsUri(String subject) {
53              // if we weren't given a pid, assume it's a URI
54              if (!isPid(subject)) {
55                  return subject;
56              }
57              // otherwise return URI from the pid
58              return toUri(subject);
59          }
60  
61          static String getSubjectPID(String subject) {
62              if (isPid(subject)) {
63                  return subject;
64              }
65              // check for info:uri scheme
66              if (subject.startsWith(FEDORA_URI_PREFIX)) {
67                  // pid is everything after the first / to the 2nd / or to the end of the string
68                  return subject.split("/", 3)[1];
69  
70              } else {
71                  throw new IllegalArgumentException(
72                          "Subject URI must be in the " + FEDORA_URI_PREFIX +
73                                  " scheme.");
74              }
75          }
76  
77          private static boolean isPid(String subject) {
78              return pidRegex.matcher(subject).matches();
79          }
80  
81          private static String toUri(String pidString) {
82              return FEDORA_URI_PREFIX + pidString;
83          }
84      }
85  
86  }
