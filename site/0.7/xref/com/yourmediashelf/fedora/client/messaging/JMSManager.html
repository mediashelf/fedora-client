115     /**
116      * Creates a JMS manager using jndi properties to start a connection
117      * to a JMS provider. If jndi properties are null an attempt will
118      * be made to create a connection based on a <code>java:comp/env</code>
119      * context
120      *
121      * A connection must have a clientId in order to create durable
122      * subscriptions. This clientId can either be set administratively
123      * on the Connection object created in the JNDI store or by providing
124      * a non-null value for the clientId parameter of this method.
125      *
126      * @param jndiProps
127      * @param clientId
128      * @throws MessagingException
129      */
130     public JMSManager(Properties jndiProps, String clientId)
131             throws MessagingException {
132 
133         this.jndiProps = jndiProps;
134 
135         // Check properties
136         if (jndiProps == null) {
137             // Attempt a lookup to see if properties are provided by the container
138             String providerUrl = null;
139             try {
140                 providerUrl = (String) jndiLookup(Context.PROVIDER_URL);
141             } catch (MessagingException me) {
142                 providerUrl = null;
143             }
144             if (providerUrl == null) {
145                 throw new MessagingException(
146                         "JMS connection properties must either be " +
147                                 "provided by the container or by a non-null properties file " +
148                                 "containing, at minimum, a value for " +
149                                 Context.INITIAL_CONTEXT_FACTORY + " and " +
150                                 Context.PROVIDER_URL + ".");
151             }
152         } else {
153             String icf = jndiProps.getProperty(Context.INITIAL_CONTEXT_FACTORY);
154             if (icf == null) {
155                 throw new MessagingException("A value for " +
156                         Context.INITIAL_CONTEXT_FACTORY +
157                         " must be included in the JNDI properties.");
158             }
159 
160             String pUrl = jndiProps.getProperty(Context.PROVIDER_URL);
161             if (pUrl == null) {
162                 throw new MessagingException("A value for " +
163                         Context.PROVIDER_URL +
164                         "must be included in the JNDI properties.");
165             }
166         }
167 
168         connectToJMS(clientId);
169     }

185     /**
186      * Creates a Destination if the Destination has not already been created.
187      *
188      * @param name - the name of the destination to create
189      * @param type - the destination type (topic or queue)
190      * @param fTransacted - determines whether the session will maintain transactions
191      * @param ackMode - determines the session acknowledgment mode
192      * @throws MessagingException
193      */
194     public Destination createDestination(String name, DestinationType type,
195             boolean fTransacted, int ackMode) throws MessagingException {
196         // If the destination already exists, just return it
197         JMSDestination jmsDest = jmsDestinations.get(name);
198         if (jmsDest != null) {
199             return jmsDest.destination;
200         }
201 
202         // Create the new destination and store it
203         Session session;
204         try {
205             session = connection.createSession(fTransacted, ackMode);
206         } catch (JMSException e) {
207             throw new MessagingException(e.getMessage(), e);
208         }
209 
210         // Look up the destination otherwise create it
211         Destination destination = null;
212         try {
213             destination = (Destination) jndiLookup(name);
214         } catch (MessagingException me) {
215             logger.debug("JNDI lookup for destination " + name + " failed. " +
216                     "Destination must be created.");
217             destination = null;
218         }
219         if (destination == null) {
220             // Create a topic or queue as specified
221             try {
222                 if (type.equals(DestinationType.Queue)) {
223                     logger.debug("setupDestination() - creating Queue" + name);
224                     destination = session.createQueue(name);
225                 } else {
226                     logger.debug("setupDestination() - creating Topic " + name);
227                     destination = session.createTopic(name);
228                 }
229             } catch (JMSException e) {
230                 throw new MessagingException(e.getMessage(), e);
231             }
232         }
233 
234         jmsDest = new JMSDestination(destination, session, null, null);
235 
236         jmsDestinations.put(name, jmsDest);
237 
238         return destination;
239     }

302             Session s =
303                     connection.createSession(false, Session.AUTO_ACKNOWLEDGE);

307             return msg;

309             throw new MessagingException(e.getMessage(), e);
310         }

444     /**
445      * This is an asynchronous and durable listen. The caller provides a JMS
446      * callback interface reference, and any messages received for the given
447      * destination are provided through the onMessage() callback method. If the
448      * listener becomes unavailable the JMS provider will store messages
449      * received on the given topic until the listener reestablishes a connection
450      * and will then deliver those messages.
451      * Messages will be filtered based on the provided message selector.
452      *
453      * @param topic
454      *        the topic on which to listen
455      * @param messageSelector
456      *        selection criteria for filtering messages
457      * @param callback
458      *        the listener to call when a message is received
459      * @param subscriptionName
460      *        the name of the subscription
461      * @return The subscription name which can be used to stop, unsubscribe,
462      *         and reconnect to this topic
463      * @throws MessagingException
464      */
465     public String listenDurable(Topic topic, String messageSelector,
466             MessageListener callback, String subscriptionName)
467             throws MessagingException {
468         try {
469             String clientId = connection.getClientID();
470             if (clientId == null) {
471                 throw new MessagingException(
472                         "A non-null client ID must be provided upon "
473                                 + "creation of a JMSManager in order to create a JMS connection "
474                                 + "capable of creating durable subscriptions.");
475             }
476             if (subscriptionName == null) {
477                 subscriptionName = topic.getTopicName();
478             }
479 
480             Session s =
481                     connection.createSession(false, Session.AUTO_ACKNOWLEDGE);
482             MessageConsumer c =
483                     s.createDurableSubscriber(topic, subscriptionName,
484                             messageSelector, false);
485             c.setMessageListener(callback);
486             durableSubscriptions.put(subscriptionName, c);
487         } catch (JMSException e) {
488             throw new MessagingException(e.getMessage(), e);
489         }
490 
491         if (logger.isDebugEnabled()) {
492             logger.debug("listen() - Asynchronous durable listen on topic " +
493                     topic);
494         }
495 
496         return subscriptionName;
497     }

570     /**
571      * Stops producers and consumers on a given destination.
572      * This has no effect on durable subscriptions.
573      *
574      * @param destName
575      * @throws MessagingException
576      */
577     public void stop(String destName) throws MessagingException {
578         try {
579             // Look for an existing destination for the given destination
580             //
581             JMSDestination jmsDest = jmsDestinations.get(destName);
582             if (jmsDest != null) {
583                 // Close out all JMS related state
584                 //
585                 if (jmsDest.producer != null) {
586                     jmsDest.producer.close();
587                     logger.debug("Closed producer for " + destName);
588                 }
589                 if (jmsDest.consumer != null) {
590                     jmsDest.consumer.close();
591                     logger.debug("Closed consumer for " + destName);
592                 }
593                 if (jmsDest.session != null) {
594                     jmsDest.session.close();
595                     logger.debug("Closed session for " + destName);
596                 }
597 
598                 jmsDest.destination = null;
599                 jmsDest.session = null;
600                 jmsDest.producer = null;
601                 jmsDest.consumer = null;
602 
603                 // Remove the JMS client entry
604                 //
605                 jmsDestinations.remove(destName);
606                 jmsDest = null;
607             }
608         } catch (JMSException e) {
609             throw new MessagingException(e.getMessage(), e);
610         }
611     }

613     /**
614      * Stops a durable message consumer. Note that this is not
615      * the same as unsubscribing. When a durable message consumer is
616      * restarted all messages received since it was stopped will be
617      * delivered.
618      *
619      * @param subscriptionName - the name of the subscription
620      * @throws MessagingException
621      */
622     public void stopDurable(String subscriptionName) throws MessagingException {
623         try {
624             MessageConsumer durableSubscriber =
625                     durableSubscriptions.get(subscriptionName);
626             if (durableSubscriber != null) {
627                 durableSubscriber.close();
628             }
629         } catch (JMSException jmse) {
630             throw new MessagingException(
631                     "Exception encountered attempting to " +
632                             "stop durable subscription with name: " +
633                             subscriptionName + ". Exception message: " +
634                             jmse.getMessage(), jmse);
635         }
636     }

638     /**
639      * Removes the durable subscription with the given name.
640      *
641      * @param subscriptionName - name of the durable subscription
642      * @throws MessagingException
643      */
644     public void unsubscribeDurable(String subscriptionName)
645             throws MessagingException {
646         try {
647             Session session =
648                     connection.createSession(false, Session.AUTO_ACKNOWLEDGE);
649             MessageConsumer consumer =
650                     durableSubscriptions.get(subscriptionName);
651             if (consumer != null) {
652                 consumer.close();
653             }
654             session.unsubscribe(subscriptionName);
655         } catch (JMSException jmse) {
656             String errMsg =
657                     "Unable to unsubscribe from subscription with name: " +
658                             subscriptionName + " due to exception: " +
659                             jmse.getMessage();
660             logger.debug(errMsg, jmse);
661             throw new MessagingException(errMsg, jmse);
662         }
663     }
