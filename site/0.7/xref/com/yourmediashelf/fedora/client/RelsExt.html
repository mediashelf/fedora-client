20  package com.yourmediashelf.fedora.client;

22  import java.io.ByteArrayOutputStream;
23  import java.io.UnsupportedEncodingException;
24  
25  import com.hp.hpl.jena.rdf.model.Model;
26  import com.hp.hpl.jena.rdf.model.ModelFactory;
27  import com.hp.hpl.jena.rdf.model.Property;
28  import com.hp.hpl.jena.rdf.model.RDFNode;
29  import com.hp.hpl.jena.rdf.model.Resource;
30  
31  /**
32   * <p>A builder for a Fedora 3.x RELS-EXT datastream.
33   * This class is only necessary for working with Fedora repositories that do not
34   * support the Relationships API (via the REST API), i.e., Fedora 3.3 and
35   * earlier.</p>
36   *
37   * <p>With Fedora 3.4.0 and later, it's recommended to use AddRelationship.</p>
38   *
39   * @author Edwin Shin
40   *
41   */
42  public class RelsExt {
43  
44      public static class Builder {
45  
46          private final String pid;
47  
48          private final Resource subject;
49  
50          private final Model model = ModelFactory.createDefaultModel();
51  
52          public Builder(String pid) {
53              if (pid.startsWith("info:fedora/")) {
54                  pid = pid.substring(12);
55              }
56              this.pid = pid;
57              model.setNsPrefix("fedora-model",
58                      "info:fedora/fedora-system:def/model#");
59              subject =
60                      model.createResource(String.format("info:fedora/%s", pid));
61              addCModel("info:fedora/fedora-system:ContentModel-3.0");
62          }
63  
64          public RelsExt build() {
65              RelsExt result = new RelsExt(pid, model);
66              return result;
67          }
68  
69          public Builder addRelationship(String predicate, String object,
70                  boolean isLiteral, String datatype) {
71              Property p = model.createProperty(predicate);
72              RDFNode o;
73              if (isLiteral) {
74                  o = model.createTypedLiteral(object, datatype);
75              } else {
76                  o = model.createResource(object);
77              }
78              model.add(subject, p, o);
79              return this;
80          }
81  
82          public Builder addCModel(String cmodel) {
83              Property p =
84                      model.createProperty("info:fedora/fedora-system:def/model#hasModel");
85              if (!cmodel.startsWith("info:fedora/")) {
86                  cmodel = "info:fedora/" + cmodel;
87              }
88              RDFNode o = model.createResource(cmodel);
89              model.add(subject, p, o);
90              return this;
91          }
92      }
93  
94      private final String pid;
95  
96      private final Model model;
97  
98      private RelsExt(String pid, Model model) {
99          this.pid = pid;
100         this.model = model;
101     }
102 
103     @Override
104     public String toString() {
105         ByteArrayOutputStream bout = new ByteArrayOutputStream();
106         model.write(bout);
107         try {
108             return bout.toString("UTF-8");
109         } catch (UnsupportedEncodingException e) {
110             throw new RuntimeException(e.getMessage(), e);
111         }
112     }
113 
114     public String getPid() {
115         return pid;
116     }
117 }
