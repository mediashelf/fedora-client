42      @Test
43      public void testTimezones() throws Exception {
44          String s1 = "2002-10-10T12:00:00-05:00";
45          DateTime dt1 = DateUtility.parseXSDDateTime(s1);
46          String s2 = "2002-10-10T17:00:00-00:00";
47          DateTime dt2 = DateUtility.parseXSDDateTime(s2);
48          assertTrue(dt1.isEqual(dt2));
49  
50          String s3 = "2002-10-10T12:00:00Z";
51          DateTime dt3 = DateUtility.parseXSDDateTime(s3);
52          assertTrue(dt2.minusHours(5).isEqual(dt3));
53      }
54  
55      @Test
56      public void testZeroYearAndEndFrag() throws Exception {
57          String s4 = "-0001-12-31T24:00:00Z";
58          DateTime dt4 = DateUtility.parseXSDDateTime(s4);
59          String s5 = "0000-01-01T00:00:00Z";
60          DateTime dt5 = DateUtility.parseXSDDateTime(s5);
61          assertTrue(dt4.isEqual(dt5));
62      }

64      @Test
65      public void testMillis() throws Exception {
66          String lex1 = "1999-12-31T22:33:44.500Z";
67          String can1 = "1999-12-31T22:33:44.5Z";
68          DateTime dt1 = DateUtility.parseXSDDateTime(lex1);
69          assertEquals(can1, DateUtility.getXSDDateTime(dt1));
70  
71          String lex2 = "1999-12-31T22:33:44.050Z";
72          String can2 = "1999-12-31T22:33:44.05Z";
73          DateTime dt2 = DateUtility.parseXSDDateTime(lex2);
74          assertEquals(can2, DateUtility.getXSDDateTime(dt2));
75  
76          String lex3 = "1999-12-31T22:33:44Z";
77          String can3 = lex3;
78          DateTime dt3 = DateUtility.parseXSDDateTime(lex3);
79          assertEquals(can3, DateUtility.getXSDDateTime(dt3));
80  
81          String lex4 = "2010-06-01T04:52:14.025Z";
82          String can4 = lex4;
83          DateTime dt4 = DateUtility.parseXSDDateTime(lex4);
84          assertEquals(can4, DateUtility.getXSDDateTime(dt4));
85  
86          String lex5 = "2010-06-01T02:52:14.25Z";
87          String can5 = "2010-06-01T02:52:14.25Z";
88          DateTime dt5 = DateUtility.parseXSDDateTime(lex5);
89          assertEquals(can5, DateUtility.getXSDDateTime(dt5));
90      }
