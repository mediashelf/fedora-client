47      @Test
48      public void testGetRelationships() throws Exception {
49          FedoraResponse response = null;
50          String subject = String.format("info:fedora/%s", testPid);
51          String predicate = "urn:foo/testGetRelationships";
52          String object = "Able was I ere I saw Elba";
53  
54          // first add a relationship
55          response =
56                  addRelationship(testPid).predicate(predicate).object(object,
57                          true).execute();
58          assertEquals(200, response.getStatus());
59  
60          // now get it
61          response = getRelationships(testPid).predicate(predicate).execute();
62          assertEquals(200, response.getStatus());
63  
64          Model model = ModelFactory.createDefaultModel();
65          model.read(response.getEntityInputStream(), null, FileUtils.langXML);
66          StmtIterator it = model.listStatements();
67          Statement s;
68          while (it.hasNext()) {
69              s = it.next();
70              assertEquals(subject, s.getSubject().toString());
71              assertEquals(predicate, s.getPredicate().toString());
72              assertEquals(object, s.getObject().toString());
73          }
74      }

106     @Test
107     public void testGetRelsIntRelationship() throws Exception {
108         String subject = String.format("info:fedora/%s/RELS-INT", testPid);
109         String predicate = "urn:foo/testGetRelsIntRelationship";
110         String object = "urn:bar/baz";
111         addRelationship(subject).predicate(predicate).object(object).execute();
112         FedoraResponse response = getRelationships(subject).execute();
113         assertEquals(200, response.getStatus());
114 
115         Model model = ModelFactory.createDefaultModel();
116         model.read(response.getEntityInputStream(), null, FileUtils.langXML);
117         StmtIterator it = model.listStatements();
118         Statement s;
119         boolean found = false;
120         while (it.hasNext()) {
121             s = it.next();
122             if (s.getSubject().toString().equals(subject) &&
123                     s.getPredicate().toString().equals(predicate) &&
124                     s.getObject().toString().equals(object)) {
125                 found = true;
126                 break;
127             }
128         }
129         assertTrue(found);
130     }
