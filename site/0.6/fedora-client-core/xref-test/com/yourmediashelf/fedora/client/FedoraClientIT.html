51  	@BeforeClass
52  	public static void setUpBeforeClass() throws Exception {
53  		String baseUrl = System.getProperty("fedora.test.baseUrl");
54  		String username = System.getProperty("fedora.test.username");
55  		String password = System.getProperty("fedora.test.password");
56  		credentials = new FedoraCredentials(new URL(baseUrl), username, password);
57  	}
58  
59  	@Before
60  	public void setUp() throws Exception {
61  		client = new FedoraClient(credentials);
62  		String debug = System.getProperty("test.debug");
63  		client.debug(Boolean.parseBoolean(debug));
64  	}
65  
66  	@Test
67  	public void testGetMimeType() throws Exception {
68  	    File f = new File("src/test/resources/testGetMimeType.xml");
69  	    assertEquals("text/xml", client.getMimeType(f));
70  	}
71  
72  	@Test
73  	public void testObjectCreationWithRelsExt() throws Exception {
74  	    // create the object
75  	    String pid = getNextPID().namespace("test").execute(client).getPid();
76  	    IngestResponse response = new Ingest(pid).execute(client);
77  	    assertEquals(pid, response.getPid());
78  
79  	    // Create RELS-EXT using RelsExt.Builder
80  	    String cmodel = "test:cmodel";
81  	    String relsExt = new RelsExt.Builder(pid)
82                                 .addCModel(cmodel).build().toString();
83  
84  	    AddDatastreamResponse addDS = new AddDatastream(pid, "RELS-EXT")
85  	                                        .content(relsExt).execute(client);
86  	    assertEquals(201, addDS.getStatus());
87  
88  	    // cleanup
89  	    FedoraResponse purge = new PurgeObject(pid).execute(client);
90  	    assertEquals(200, purge.getStatus());
91  	}
